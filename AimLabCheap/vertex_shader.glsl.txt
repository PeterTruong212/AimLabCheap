#version 330 core
layout (location = 0) in vec3 aPos;    // Vị trí của đỉnh
layout (location = 1) in vec3 aNormal; // Pháp tuyến

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec3 fragNormal;  // Truyền pháp tuyến tới fragment shader
out vec3 fragPos;     // Truyền vị trí tới fragment shader

void main()
{
    fragPos = vec3(model * vec4(aPos, 1.0)); // Tính toán vị trí của đỉnh trong không gian world
    fragNormal = mat3(transpose(inverse(model))) * aNormal; // Pháp tuyến đã được chuyển đổi

    // Ghi vào gl_Position để xác định vị trí đỉnh trong không gian clip
    gl_Position = projection * view * vec4(fragPos, 1.0);
}
